\section{Descripción del Problema}

Se cuenta con datos de 55 familias de la India sobre su \textbf{Gasto en Comida}
(variable dependiente, $Y$) y su \textbf{Gasto Total} (variable independiente, $X$),
ambas en rupias. El modelo de regresión es:
\[
  \text{Gasto\_en\_Comida}_i = \beta_0 + \beta_1 \cdot \text{Gasto\_Total}_i + u_i,
  \quad i = 1, 2, \ldots, 55
\]

\section{Inciso C: Pruebas Formales de Heteroscedasticidad}

%---------------------------------------------------------------
\subsection{Prueba de Breusch-Pagan / Cook-Weisberg}

\subsubsection{Hipótesis y estadístico}
\[
  H_0: \text{Var}(u_i \mid X_i) = \sigma^2 \qquad
  H_a: \text{Var}(u_i \mid X_i) = f(X_i)
\]
\[
  \text{LM} = n \cdot R^2_{\text{aux}} \;\sim\; \chi^2(p) \quad \text{bajo } H_0
\]
donde la \textbf{regresión auxiliar} es $\hat{u}^2_i = \alpha_0 + \alpha_1 \hat{Y}_i + v_i$
y $p=1$ regresor.

\subsubsection{Proceso completo con $F_c$ y $\chi^2_c$}

\begin{cajacomando}[title=Stata -- Breusch-Pagan]
\texttt{. estat hettest} \quad \textit{\% también: reg R yhat, luego calcular LM = n*R2}
\end{cajacomando}

\begin{table}[H]
\centering
\caption{Proceso de cálculo: Breusch-Pagan}
\begin{tabular}{llr}
\toprule
\textbf{Paso} & \textbf{Operación} & \textbf{Valor}\\
\midrule
1 & $n$ (observaciones) & 55 \\
2 & $R^2_{\text{aux}}$ (reg.\ aux.\ $\hat{u}^2$ sobre $\hat{Y}$) & 0.1305 \\
3 & $g$ (n.º regresores en reg.\ aux.) & 1 \\
\midrule
4 & $\chi^2_c = n \cdot R^2_{\text{aux}} = 55 \times 0.1305$ & \textbf{7.18} \\
5 & $F_c = \dfrac{R^2_{\text{aux}}/g}{(1-R^2_{\text{aux}})/(n-g-1)}

\midrule
