\section{Descripción del Problema}

Se cuenta con datos de 55 familias de la India sobre su \textbf{Gasto en Comida}
(variable dependiente, $Y$) y su \textbf{Gasto Total} (variable independiente, $X$),
ambas medidas en rupias. El objetivo es estudiar la relación entre ambas variables,
diagnosticar y corregir el problema de \textbf{heteroscedasticidad}.

El modelo de regresión es:
\[
  \text{Gasto\_en\_Comida}_i = \beta_0 + \beta_1 \cdot \text{Gasto\_Total}_i + u_i,
  \quad i = 1, 2, \ldots, 55
\]

\begin{itemize}
  \item \textbf{Intercepto ($\hat{\beta}_0 = 94.21$):} Cuando el gasto total es cero,
        el gasto estimado en comida sería de 94.21 rupias (sin interpretación económica
        directa, ya que no se puede tener gasto total igual a cero en la práctica).
  \item \textbf{Pendiente ($\hat{\beta}_1 = 0.4368$):} Por cada rupia adicional de
        gasto total, el gasto en comida aumenta en promedio \textbf{0.4368 rupias}.

  \item \textbf{Significancia:} El coeficiente de Gasto\_Total es estadísticamente

        datos de corte transversal.
  \item \textbf{Ecuación estimada:}
\end{itemize}
\[
  \widehat{\text{Gasto\_en\_Comida}} = 94.2088 + 0.4368 \cdot \text{Gasto\_Total}

\section{Inciso C: Pruebas Formales de Heteroscedasticidad}

\subsection{Prueba de Breusch-Pagan / Cook-Weisberg (LM)}

\subsubsection{Hipótesis Formales}
\[
  H_0: \text{Var}(u_i \mid X_i) = \sigma^2
  \quad \text{(varianza constante — homocedasticidad)}
\]
\[
  H_a: \text{Var}(u_i \mid X_i) = f(X_i)
  \quad \text{(varianza depende de } X \text{ — heteroscedasticidad)}
\]

\textbf{Estadístico:} $\text{LM} = n \cdot R^2_{\text{aux}} \sim \chi^2(p)$ bajo $H_0$.\\
\textbf{Regla de decisión:} Rechazar $H_0$ si p-valor $< \alpha$.

\subsubsection{Comando en Stata}

\begin{cajacomando}[title=Stata -- Prueba Breusch-Pagan]
\texttt{. estat hettest}
\end{cajacomando}

\subsubsection{Resultados}

\begin{table}[H]
\centering
\caption{Prueba Breusch-Pagan/Cook-Weisberg}
\begin{tabular}{lcc}
\toprule
\textbf{Hipótesis nula} & \multicolumn{2}{c}{$H_0$: Varianza constante (homocedasticidad)}\\
\midrule
$\chi^2(1)$     & = & 7.18\\
Prob $> \chi^2$ & = & \textbf{0.0074}\\
\bottomrule
\end{tabular}
\end{table}

\subsubsection{Interpretación}

\begin{cajaconclucion}[title=Conclusión -- Breusch-Pagan]
El estadístico $\chi^2(1) = 7.18$ con p-valor $= 0.0074 < 0.01$.
Se \textbf{rechaza la hipótesis nula de varianza constante} al 1\% de significancia.
La prueba confirma formalmente la presencia de \textbf{heteroscedasticidad} en el
modelo original.
\end{cajaconclucion}

\subsection{Prueba de White}

\subsubsection{Hipótesis Formales}
\[
  H_0: \text{Var}(u_i \mid X_i) = \sigma^2 \quad \text{(homocedasticidad)}
\]
\[
  H_a: \text{Var}(u_i) \text{ depende de } X_i, X_i^2
  \text{ y/o sus productos cruzados}
\]

\end{itemize}

\subsection{Prueba de Park}

\subsubsection{Fundamento Teórico}

Park (1966) propone que $\sigma_i^2 = \sigma^2 X_i^{\beta} e^{v_i}$. Tomando logaritmos:
\[
  \ln(\hat{u}_i^2) = \alpha_0 + \alpha_1 \ln(X_i) + v_i
\]
Si $\alpha_1$ es significativo, existe heteroscedasticidad.

\subsubsection{Comandos en Stata}

\begin{cajacomando}[title=Stata -- Prueba de Park]
\texttt{. gen ln\_u2 = ln(errores\^{}2)}\\
\texttt{. gen ln\_x  = ln(Gasto\_Total)}\\
\texttt{. reg ln\_u2 ln\_x}\\
\texttt{. twoway (scatter ln\_u2 ln\_x) (lfit ln\_u2 ln\_x), ///}\\
\texttt{\ \ \ \ xtitle("ln(Gasto Total)") ytitle("ln(u\^{}2)")}
\end{cajacomando}

\subsection{Prueba de Glejser}

\subsubsection{Fundamento Teórico}

Glejser (1969) propone regresar $|\hat{u}_i|$ sobre transformaciones de $X_i$:
\begin{enumerate}

\subsection{Prueba de Goldfeld-Quandt}

\subsubsection{Fundamento Teórico}

Goldfeld-Quandt (1965): ordenar por $X$, eliminar $c$ observaciones del centro,
estimar MCO por subgrupo y calcular:
\[
  \lambda = \frac{\text{SCR}_2 / (m-k)}{\text{SCR}_1 / (m-k)}
          = \frac{\text{SCR}_2}{\text{SCR}_1} \sim F(m-k,\; m-k)
\]

\subsubsection{Hipótesis Formales}
\[
  H_0: \sigma_1^2 = \sigma_2^2 \qquad
  H_a: \sigma_2^2 > \sigma_1^2
\]

\section{Inciso D: Corrección mediante Transformación Logarítmica}

\subsection{Justificación Teórica}

El modelo log-log estabiliza la varianza cuando $\sigma_i^2 \propto X_i$:
\[
  \ln(\text{Gasto\_en\_Comida}_i) = \beta_0 + \beta_1 \ln(\text{Gasto\_Total}_i) + u_i
\]
donde $\beta_1$ es la \textbf{elasticidad-gasto en comida}.

\subsection{Comandos en Stata}

\begin{cajacomando}[title=Stata -- Transformación Logarítmica]
\texttt{. gen logaritmo\_y = ln(Gasto\_en\_Comida)}\\
\texttt{. gen logaritmo\_x = ln(Gasto\_Total)}\\
\texttt{. reg logaritmo\_y logaritmo\_x}\\
\texttt{. estat hettest}\\
\texttt{. predict errores\_log, residuals}\\
\texttt{. rvfplot}\\
\texttt{. kdensity errores\_log, normal}
\end{cajacomando}

\begin{enumerate}
  \item \textbf{Elasticidad ($\hat{\beta}_1 = 0.7363$):} Por cada 1\% de aumento en
        el gasto total, el gasto en comida aumenta \textbf{0.7363\%}. Los alimentos se
        comportan como \textbf{bien normal no de lujo} (Ley de Engel: la proporción
        del ingreso destinada a alimentos disminuye conforme el ingreso crece).

  \item \textbf{Mejora en el ajuste:} $R^2$ mejora de 0.3698 (lineal) a \textbf{0.4125}
        (log-log).

  \item \textbf{Corrección de heteroscedasticidad:} Breusch-Pagan post-corrección:
        $\chi^2(1) = 1.60$, p $= 0.2052 > 0.05$. \textbf{No se rechaza $H_0$}.
        La transformación logarítmica \textbf{corrigió exitosamente} el problema.

  \item \textbf{Ecuación estimada:}
  \[
    \widehat{\ln(\text{GC})} = 1.1543 + 0.7363 \cdot \ln(\text{GT})
    \quad \Leftrightarrow \quad
    \widehat{\text{GC}} = e^{1.1543} \cdot \text{GT}^{0.7363}
  \]
\end{enumerate}
\end{cajaconclucion}

\section{Inciso E: Errores Estándar Robustos de White}

\subsection{Fundamento Teórico}

White (1980) propone estimar la varianza de $\hat{\beta}$ de forma consistente:
\[
  \widehat{\text{Var}}_{HC}(\hat{\boldsymbol{\beta}})
  = (X'X)^{-1} \left(\sum_{i=1}^{n} \hat{u}_i^2 x_i x_i'\right) (X'X)^{-1}
\]
