\section{Enunciado del Problema 10.1}

\begin{enunciado}[title=Enunciado 10.1 --- Gujarati \& Porter]

normales para estimar las $k$ inc\'ognitas.  Suponga que $X_k$ es una
combinaci\'on lineal perfecta de las variables $X$ restantes.
\textbf{?`C\'omo se demostrar\'ia que en este caso es imposible estimar
los $k$ coeficientes de regresi\'on?}\\[3pt]

\end{enunciado}

\section{Marco Te\'orico}

\subsection{Estimador MCO Matricial}

El modelo $\bm{Y}=\bm{X\beta}+\bm{u}$ admite estimaci\'on MCO \'unica si y
solo si $\rg(\bm{X})=k$.

\begin{definicion}
  Bajo los supuestos de Gauss-Markov ($\E[\bm{u}|\bm{X}]=\bm{0}$,
  $\Var[\bm{u}|\bm{X}]=\sigma^2\bm{I}_n$, $\rg(\bm{X})=k$), el estimador
  MCO es:
  \[
    \hat{\bm{\beta}} = (\bm{X}'\bm{X})^{-1}\bm{X}'\bm{Y}
  \]
  El supuesto $\rg(\bm{X})=k$ es \emph{indispensable} para que
  $(\bm{X}'\bm{X})^{-1}$ exista.
\end{definicion}

\subsection{Las $k$ Ecuaciones Normales}

La condici\'on de primer orden $\partial\,\text{SRC}/\partial\bm{\beta}=\bm{0}$
produce $\bm{X}'\bm{X}\hat{\bm{\beta}}=\bm{X}'\bm{Y}$.  En desviaciones
respecto a la media, con dos regresores:

\begin{equation}
  \begin{pmatrix} \sxdos & \sxdxt \\ \sxdxt & \sxtres \end{pmatrix}
  \begin{pmatrix}\hat{\beta}_2\\\hat{\beta}_3\end{pmatrix}
  =\begin{pmatrix}\sxdy\\\sxty\end{pmatrix}
  \label{eq:sistema2}
\end{equation}

\subsection{Detecci\'on Emp\'irica}

Equivalentemente, sobre las columnas de $\bm{X}$:

con $(\lambda_1,\lambda_2,\lambda_3)=(-1,-2,1)\neq\bm{0}$.
\end{alerta}

\subsection{Reformulaci\'on en Desviaciones}

Las medias son $\bar{Y}=0$, $\bar{X}_2=6$, $\bar{X}_3=11$.
En desviaciones: $x_{2i}=X_{2i}-6$, $x_{3i}=X_{3i}-11$.  Dado que
$X_{3i}=2X_{2i}-1$:
\[
  x_{3i}=(2X_{2i}-1)-11=2(X_{2i}-6)=2x_{2i}
\]

\begin{resultado}[title=Relaci\'on en desviaciones]
\[
  x_{3i}=2x_{2i}, \quad \forall\,i \qquad (\lambda=2)
\]
\end{resultado}

\section{Demostraci\'on: Imposibilidad de Estimaci\'on}

\subsection{Proposici\'on Central}

\begin{prop}[Indeterminaci\'on bajo Multicolinealidad Perfecta]
Si existe $\bm{c}\neq\bm{0}$ tal que $\bm{Xc}=\bm{0}$, entonces
$\hat{\bm{\beta}}=(\bm{X}'\bm{X})^{-1}\bm{X}'\bm{Y}$ \textbf{no existe}.
\end{prop}

\begin{dem}
\textbf{Paso 1. Singularidad de $\bm{X}'\bm{X}$.}
Si $\bm{Xc}=\bm{0}$, premultiplicando por $\bm{X}'$:
\[
  (\bm{X}'\bm{X})\bm{c}=\bm{0},\quad \bm{c}\neq\bm{0}
  \;\Longrightarrow\; \det(\bm{X}'\bm{X})=0
\]

\textbf{Paso 2. Inversa indefinida.}
$\det(\bm{X}'\bm{X})=0 \Rightarrow (\bm{X}'\bm{X})^{-1}$ no existe.

\textbf{Paso 3. Infinitas soluciones.}
Sea $\hat{\bm{\beta}}^*$ una soluci\'on particular.  Para todo $\alpha\in\mathbb{R}$:
\[
  \bm{X}'\bm{X}(\hat{\bm{\beta}}^*+\alpha\bm{c})
  =\bm{X}'\bm{Y}+\alpha\underbrace{(\bm{X}'\bm{X})\bm{c}}_{=\bm{0}}
  =\bm{X}'\bm{Y}
\]
Existe una \textbf{familia infinita} de soluciones: los coeficientes
individuales son inidentificables.\hfill$\square$
\end{dem}

\subsection{V\'ia Rango y el Teorema de la Dimensi\'on}

\begin{teorema}[title=Rango de $\bm{X}'\bm{X}$]
Para toda matriz $\bm{X}$: $\rg(\bm{X}'\bm{X})=\rg(\bm{X})$.
\end{teorema}

Columnas linealmente dependientes $\Rightarrow \rg(\bm{X})<k \Rightarrow
\rg(\bm{X}'\bm{X})<k \Rightarrow \det(\bm{X}'\bm{X})=0$.

\subsection{V\'ia Desigualdad de Cauchy-Schwarz}

\begin{equation}
  D=\det(\bm{M})=\left(\sxdos\right)\!\left(\sxtres\right)-\left(\sxdxt\right)^2
\end{equation}

La igualdad de Cauchy-Schwarz se alcanza $\Leftrightarrow$ $x_{3i}=\lambda x_{2i}$
(proporcionalidad exacta):

\begin{alerta}
Cuando $x_{3i}=\lambda x_{2i}$:
\[
  D=\lambda^2\!\left(\sxdos\right)^2-\lambda^2\!\left(\sxdos\right)^2=0
\]
$\bm{M}$ es singular: \textbf{estimadores MCO indeterminados.}
\end{alerta}

\section{Verificaci\'on Num\'erica}

\subsection{C\'alculo del Determinante}

\begin{align}
  D &= (110)(440)-(220)^2 \nonumber\\
    &= 48{,}400-48{,}400 = \boxed{0}
\end{align}

Directamente con $\lambda=2$:
\[
  \sum x_{3i}^2=4\times110=440\;\checkmark,\quad
  \sum x_{2i}x_{3i}=2\times110=220\;\checkmark
\]

\subsection{Coeficiente de Correlaci\'on $r_{23}$}

\begin{equation}
  r_{23}=\frac{220}{\sqrt{110\times440}}=\frac{220}{220}=\mathbf{1}
\end{equation}

\subsection{Regla de Cramer: Forma $0/0$}

Aplicando la Regla de Cramer:
\begin{align}
  \hat{\beta}_2
  &=\frac{(220)(440)-(440)(220)}{0}
  =\frac{0}{0} \\[4pt]
  \hat{\beta}_3
  &=\frac{(440)(110)-(220)(220)}{0}
  =\frac{0}{0}
\end{align}

\begin{alerta}
$0/0$ es una \textbf{indeterminaci\'on matem\'atica}: no existe ning\'un
m\'etodo aritm\'etico que la resuelva.  El estimador MCO es
\textbf{algebraicamente imposible}.
\end{alerta}

\subsection{Infinitas Soluciones}

La primera ecuaci\'on normal se reduce a:
\[
  110\hat{\beta}_2+220\hat{\beta}_3=220
  \;\Leftrightarrow\;
  \hat{\beta}_2+2\hat{\beta}_3=2
\]
Su soluci\'on general, parametrizada por $\alpha\in\mathbb{R}$:
\begin{equation}
  \hat{\beta}_2=2-2\alpha,\quad \hat{\beta}_3=\alpha
\end{equation}

\begin{center}
\small
\begin{tabular}{ccc}
  \toprule
  \rowcolor{accentblue!15}
  $\alpha$ & $\hat{\beta}_2$ & $\hat{\beta}_3$ \\
  \midrule
  $0$   & $2$  & $0$ \\
  $1$   & $0$  & $1$ \\
  $1/2$ & $1$  & $1/2$ \\
  $-1$  & $4$  & $-1$ \\
  \bottomrule
\end{tabular}
\end{center}

\begin{corolario}
Todos los pares $(2-2\alpha,\,\alpha)$ producen el \textbf{mismo ajuste}:
\[
  \hat{y}_i=(2-2\alpha)x_{2i}+\alpha(2x_{2i})=2x_{2i}
\]
La funci\'on ajustada es \'unica; los coeficientes individuales, no.
\end{corolario}

\section{Consecuencias Estad\'isticas}

\subsection{Varianzas Infinitas}

\begin{equation}
  \Var(\hat{\beta}_2)
  =\frac{\sigma^2}{\sxdos\,(1-r_{23}^2)}
  =\frac{\sigma^2}{110\times0}
  \longrightarrow+\infty
\end{equation}

\subsection{An\'alisis Espectral de la Matriz M}

Ecuaci\'on caracter\'istica: $\mu(\mu-550)=0$.
Eigenvalores: $\mu_1=0$, $\mu_2=550$.

\begin{resultado}[title=Interpretaci\'on Espectral]
$\mu_1=0$ confirma singularidad: $\det(\bm{M})=0\times550=0$,
$\rg(\bm{M})=1<2$.  N\'umero de condici\'on:
$\kappa=550/0=+\infty$ (se\~nal definitiva).
\end{resultado}

\subsection{Las Tres Consecuencias Formales}

\begin{enumerate}[leftmargin=*,itemsep=1pt,label=\textbf{\arabic*.}]
  \item \textbf{Indeterminaci\'on:} $(\bm{X}'\bm{X})^{-1}$ indefinida; infinitas soluciones.
  \item \textbf{Varianzas infinitas:} $\Var(\hat{\beta}_j)\to+\infty$; intervalos $(-\infty,+\infty)$.
  \item \textbf{Fallo de identificaci\'on:} imposible separar efecto de $X_2$ y $X_3$.
\end{enumerate}

\section{Generalizaci\'on al Modelo de $k$ Variables}

\begin{prop}[Caso General]
Si $X_k=\sum_{j=1}^{k-1}\lambda_j X_j$ con alg\'un $\lambda_j\neq0$, entonces:
(a) columnas de $\bm{X}$ son linealmente dependientes;
(b) $\rg(\bm{X})<k$ y $\rg(\bm{X}'\bm{X})<k$;
(c) $\det(\bm{X}'\bm{X})=0$ e $(\bm{X}'\bm{X})^{-1}$ no existe;
(d) infinitas soluciones a las $k$ ecuaciones normales;
(e) $\Var(\hat{\beta}_j)\to+\infty$ para todo $j$ afectado.
\end{prop}

Si $X_k$ es combinaci\'on lineal de las dem\'as, la \'ultima fila de la
matriz de coeficientes normales es combinaci\'on lineal de las otras filas,
haciendo el sistema \textbf{compatible indeterminado} en vez de
\textbf{compatible determinado} (requerido para MCO).

\begin{resultado}[title=Conclusi\'on Problema 10.1]
La multicolinealidad perfecta $X_{3i}=2X_{2i}-1$ implica
$D=48{,}400-48{,}400=0$, $r_{23}=1$, eigenvalores $\{0,550\}$,
y formas $0/0$ para $\hat{\beta}_2$ y $\hat{\beta}_3$.
La estimaci\'on MCO individual es
\textbf{algebraica y estad\'isticamente imposible.}
\end{resultado}

{\large\bfseries\scshape\color{harvard} Problema 10.2}\\[1pt]
{\small\itshape\color{voguegold} Funciones Estimables bajo Multicolinealidad}
\end{center}
\vspace{4pt}

\section{Enunciado del Problema 10.2}

\begin{enunciado}[title=Enunciado 10.2 --- Gujarati \& Porter]
Considere el mismo conjunto de datos del Problema~10.1.  Se desea ajustar:
\[
  Y_i=\beta_1+\beta_2 X_{2i}+\beta_3 X_{3i}+u_i
\]
\textbf{(a)} ?`Puede estimar las tres inc\'ognitas $\beta_1,\beta_2,\beta_3$?
?`Por qu\'e?\\[3pt]
\textbf{(b)} Si no se puede, ?`qu\'e \emph{funciones lineales} de estos
par\'ametros (funciones estimables) s\'i pueden estimarse?
Muestre los c\'alculos necesarios.
\end{enunciado}

\section{Parte (a): Imposibilidad de Estimaci\'on Individual}

\lettrine[lines=2]{\color{harvard}\textbf{D}}{el} an\'alisis del Problema~10.1
sabemos que los datos satisfacen la relaci\'on de dependencia perfecta:
\[
  \bm{X}_3=2\bm{X}_2-\bm{X}_1
  \;\Longleftrightarrow\;
  -\bm{X}_1-2\bm{X}_2+\bm{X}_3=\bm{0}
\]

\begin{alerta}
\textbf{No}, no es posible estimar individualmente $\beta_1,\beta_2,\beta_3$.
La matriz $\bm{X}_{11\times3}$ tiene rango 2 (en vez de 3),
$\det(\bm{X}'\bm{X})=0$, $(\bm{X}'\bm{X})^{-1}$ no existe y las ecuaciones
normales tienen infinitas soluciones.
\end{alerta}

\subsection{Consecuencia Geom\'etrica}

El espacio columna de $\bm{X}$ es bidimensional:
\[
  \text{col}(\bm{X})=\text{span}\{\bm{X}_1,\bm{X}_2,\bm{X}_3\}
  =\text{span}\{\bm{X}_1,\bm{X}_2\}
\]
El modelo tiene tres par\'ametros pero solo \textbf{dos grados de libertad
param\'etricos} efectivos.

\section{Parte (b): Funciones Estimables}

\subsection{Marco Te\'orico}

\begin{definicion}
Una combinaci\'on lineal $\bm{\lambda}'\bm{\beta}$ es \textbf{estimable}
si existe $\bm{a}$ tal que $\bm{a}'\bm{Y}$ es un estimador lineal insesgado
de $\bm{\lambda}'\bm{\beta}$.  Equivalentemente, $\bm{\lambda}\in\text{row}(\bm{X})$.
\end{definicion}

\subsection{Derivaci\'on Algebraica por Sustituci\'on}

\textbf{Paso 1.} Modelo original:
\[
  Y_i=\beta_1+\beta_2 X_{2i}+\beta_3 X_{3i}+u_i
\]

\textbf{Paso 2.} Sustituci\'on de $X_{3i}=2X_{2i}-1$:
\begin{align}
  Y_i &=\beta_1+\beta_2 X_{2i}+\beta_3(2X_{2i}-1)+u_i \nonumber\\
      &=(\beta_1-\beta_3)+(\beta_2+2\beta_3)X_{2i}+u_i
\end{align}

\textbf{Paso 3.} Modelo reducido:
\begin{equation}
  Y_i=\underbrace{(\beta_1-\beta_3)}_{\alpha_1}
     +\underbrace{(\beta_2+2\beta_3)}_{\alpha_2}\,X_{2i}+u_i
  \label{eq:reducido}
\end{equation}

\begin{resultado}[title=Funciones Estimables Identificadas]
Las dos \'unicas combinaciones lineales independientes estimables son:
\begin{align}
  \alpha_1 &= \beta_1-\beta_3 \quad\text{(intercepto reducido)} \\
  \alpha_2 &= \beta_2+2\beta_3 \quad\text{(pendiente reducida)}
\end{align}
\end{resultado}

canal causal.

S\'i vale la pena a\~nadir la TFT.  Los cuatro criterios ---estad\'istico
$t$, $\bar{R}^2$, criterios de informaci\'on y teor\'ia econ\'omica---
convergen en favorecer su inclusi\'on.
\end{corolario}

\section{Parte (c): Significancia Individual y Colinealidad}

\lettrine[lines=2]{\color{harvard}\textbf{N}}{o.} La afirmaci\'on de que
``todos los coeficientes son significativos, luego no hay colinealidad'' es
un \textbf{error conceptual cl\'asico}.

\subsection{El Factor de Inflaci\'on de la Varianza}

\begin{definicion}
El \textbf{FIV} del estimador $\hat{\beta}_j$ es:
\[
  \text{FIV}_j=\frac{1}{1-R_j^2},\quad
  \Var(\hat{\beta}_j)=\frac{\hat{\sigma}^2}{\sum x_{ji}^2}\cdot\text{FIV}_j
\]
donde $R_j^2$ es el $R^2$ de la regresi\'on auxiliar de $X_j$ sobre
los dem\'as regresores.
\end{definicion}

La colinealidad infla el denominador del estad\'istico $t$:
\[
  t_j=\frac{\hat{\beta}_j}{\sqrt{\hat{\sigma}^2\cdot[(\bm{X}'\bm{X})^{-1}]_{jj}}}
\]
Si los $t$ siguen siendo significativos pese a ello, simplemente indica
que la colinealidad \textbf{no es lo suficientemente severa} para anularlos.

\subsection{Evidencia Indirecta de Colinealidad}

El cambio en $\hat{\beta}_{\text{TAM}}$ al incluir TFT:
\begin{equation}
  \frac{|\Delta\hat{\beta}_{\text{TAM}}|}{|\hat{\beta}_{\text{TAM}}^{\text{orig}}|}
  =\frac{0.4636}{2.2316}\approx20.8\%
\end{equation}

Un cambio de esta magnitud indica alta correlaci\'on entre TAM y TFT.
Formalmente, si $r_{\text{TAM,TFT}}$ es elevada, entonces $R_{\text{TAM}}^2$
en la regresi\'on auxiliar es alto y $\text{FIV}_{\text{TAM}}\gg1$.

\begin{alerta}
\textbf{Regla de Klein (1962):} si $R_j^2$ (regresi\'on auxiliar) supera
$R^2$ del modelo principal, la colinealidad es problem\'atica.
La colinealidad existe y es econom\'etricamente relevante,
aunque no impide la estimaci\'on.
\end{alerta}

\subsection{La Colinealidad es una Cuesti\'on de Grado}

La multicolinealidad perfecta ($\det=0$) hace inviable la estimaci\'on.
La imperfecta ($0<|\det|<\infty$) no la impide, pero:

\begin{enumerate}[leftmargin=*,itemsep=1pt,label=\textit{\arabic*.}]
  \item Infla errores est\'andar, reduciendo la potencia de los tests.
  \item Hace los estimadores sensibles a cambios muestrales o de especificaci\'on.
  \item Dificulta atribuir el efecto marginal a una sola variable.
\end{enumerate}

\begin{resultado}[title=Conclusi\'on Problema 10.3]
\textbf{(a)} $R^2\uparrow$ (+3.97\%); $\hat{\beta}_{\text{TAM}}$ menos
negativo por correcci\'on del sesgo OVB.\\[3pt]
\textbf{(b)} S\'i, la TFT debe incluirse: $t=3.07$ ($p=0.003$),
$\bar{R}^2\uparrow$, sustento te\'orico s\'olido.\\[3pt]
\textbf{(c)} No es posible concluir ausencia de colinealidad.
El cambio en $\hat{\beta}_{\text{TAM}}$ es evidencia directa de
correlaci\'on entre regresores.  La colinealidad imperfecta existe y
afecta las estimaciones sin hacerlas inviables.
\end{resultado}

%----------------------------------------------------------
\subsection*{A.1\ \ Modelo de Regresi\'on Lineal M\'ultiple}
%----------------------------------------------------------

El modelo poblacional con $k-1$ regresores e intercepto:
\[
  Y_i=\beta_1+\sum_{j=2}^{k}\beta_j X_{ji}+u_i,\quad i=1,\ldots,n
\]
En matricial: $\bm{y}_{n\times1}=\bm{X}_{n\times k}\bm{\beta}_{k\times1}+\bm{u}_{n\times1}$.

\textbf{Supuestos de Gauss-Markov:}
\begin{enumerate}[leftmargin=*,itemsep=1pt,label=\textit{\Roman*.}]
  \item $\E[\bm{u}|\bm{X}]=\bm{0}$ (media condicional nula)
  \item $\Var[\bm{u}|\bm{X}]=\sigma^2\bm{I}_n$ (homocedasticidad)
  \item $\rg(\bm{X})=k$ (no multicolinealidad perfecta)
\end{enumerate}

\textit{Interpretaci\'on econ\'omica:} cada $\beta_j$ mide el efecto
parcial de $X_j$ sobre $Y$ manteniendo fijos los dem\'as regresores
(principio \emph{ceteris paribus}).

%----------------------------------------------------------
\subsection*{A.2\ \ Estimador MCO y sus Propiedades}
%----------------------------------------------------------

\[
  \hat{\bm{\beta}}=(\bm{X}'\bm{X})^{-1}\bm{X}'\bm{y}
\]

\textbf{Propiedades:}
\begin{itemize}[leftmargin=*,itemsep=1pt]
  \item \textit{Linealidad:} $\hat{\bm{\beta}}=\bm{Ay}$, $\bm{A}=(\bm{X}'\bm{X})^{-1}\bm{X}'$
  \item \textit{Insesgadez:} $\E[\hat{\bm{\beta}}|\bm{X}]=\bm{\beta}$
  \item \textit{Eficiencia (BLUE):} m\'inima varianza entre estimadores
    lineales insesgados (Gauss-Markov)
\end{itemize}

\textit{Geometr\'ia:} $\bm{X}'\bm{y}$ proyecta $\bm{y}$ sobre el espacio
columna de $\bm{X}$; $(\bm{X}'\bm{X})^{-1}$ normaliza por la dispersi\'on
de los regresores.

%----------------------------------------------------------
\subsection*{A.3\ \ Descomposici\'on de la Varianza}
%----------------------------------------------------------

\[
  \underbrace{\bm{y}'\bm{y}-n\bar{y}^2}_{\text{STC}}
  =\underbrace{\hat{\bm{\beta}}'\bm{X}'\bm{y}-n\bar{y}^2}_{\text{SEC}}
  +\underbrace{\hat{\bm{u}}'\hat{\bm{u}}}_{\text{SRC}}
\]

\begin{center}
\small
\begin{tabular}{lll}
  \toprule
  \rowcolor{softblue}
  Sigla & Nombre & Significado \\
  \midrule
  STC & Total & Variabilidad total de $Y$ \\
  SEC & Explicada & Explicada por $\bm{X}$ \\
  SRC & Residual & No explicada \\
  \bottomrule
\end{tabular}
\end{center}

%----------------------------------------------------------
\subsection*{A.4\ \ Varianza Residual y Errores Est\'andar}
%----------------------------------------------------------

\[
  \hat{\sigma}^2=\frac{\hat{\bm{u}}'\hat{\bm{u}}}{n-k}=\frac{\text{SRC}}{n-k}
\]
\[
  \Var(\hat{\bm{\beta}})=\hat{\sigma}^2(\bm{X}'\bm{X})^{-1}
\]
\[
  \text{ee}(\hat{\beta}_j)=\sqrt{\hat{\sigma}^2\cdot[(\bm{X}'\bm{X})^{-1}]_{jj}}
\]

El denominador $n-k$ corrige los grados de libertad consumidos en estimar
$k$ par\'ametros.

\textit{Interpretaci\'on:} $\hat{\sigma}=39.13$ (Prob.~10.3) implica un
error promedio de 39 muertes por $1{,}000$ nacidos vivos al predecir MI.

%----------------------------------------------------------
\subsection*{A.5\ \ Coeficiente de Determinaci\'on $R^2$}
%----------------------------------------------------------

\[
  R^2=\frac{\text{SEC}}{\text{STC}}=1-\frac{\text{SRC}}{\text{STC}}
  =\frac{\hat{\bm{\beta}}'\bm{X}'\bm{y}-n\bar{y}^2}{\bm{y}'\bm{y}-n\bar{y}^2}
\]

\[
  \bar{R}^2=1-\frac{\text{SRC}/(n-k)}{\text{STC}/(n-1)}
  =1-(1-R^2)\frac{n-1}{n-k}
\]

\textbf{Regla de inclusi\'on:} a\~nadir $X_j$ aumenta $\bar{R}^2$
$\Leftrightarrow$ $|t_j|>1$.

\textit{Interpretaci\'on:} $R^2=0.7474$ (Prob.~10.3) implica que el
modelo explica el $74.74\%$ de la variabilidad de MI.

%----------------------------------------------------------
\subsection*{A.6\ \ Estad\'istico $t$ y Test Individual}
%----------------------------------------------------------

Para $H_0:\beta_j=0$ vs $H_1:\beta_j\neq0$:
\[
  t_j=\frac{\hat{\beta}_j}{\text{ee}(\hat{\beta}_j)}\;\sim\;t_{n-k}\quad
  \text{bajo }H_0
\]

Se rechaza $H_0$ si $|t_j|>t_{\alpha/2,\,n-k}$.

Para TFT: $t=12.869/4.191=3.071$, $p=0.0032<0.05$.

\textit{Interpretaci\'on:} si $\beta_4=0$ en la poblaci\'on, la
probabilidad de observar $|t|\geq3.07$ es solo $0.32\%$: evidencia
muy fuerte de efecto real.

%----------------------------------------------------------
\subsection*{A.7\ \ Estad\'istico $F$ y Test Global}
%----------------------------------------------------------

Para $H_0:\beta_2=\cdots=\beta_k=0$:
\[
  F=\frac{\text{SEC}/(k-1)}{\text{SRC}/(n-k)}
  =\frac{R^2/(k-1)}{(1-R^2)/(n-k)}\;\sim\;F_{k-1,\,n-k}
\]

En Prob.~10.3: $F=59.17$ con $p<0.0001$.

\textit{Interpretaci\'on:} PIBPC, TAM y TFT explican conjuntamente MI
de forma altamente significativa.

%----------------------------------------------------------
\subsection*{A.8\ \ Sesgo por Omisi\'on de Variable (OVB)}
%----------------------------------------------------------

Modelo verdadero: $\bm{y}=\bm{X}_1\bm{\beta}_1+\bm{X}_2\bm{\beta}_2+\bm{u}$.
Si se estima omitiendo $\bm{X}_2$:
\[
  \plim\,\hat{\bm{\beta}}_1^{\text{corto}}
  =\bm{\beta}_1+(\bm{X}_1'\bm{X}_1)^{-1}\bm{X}_1'\bm{X}_2\,\bm{\beta}_2
\]

El sesgo es nulo solo si $\bm{\beta}_2=\bm{0}$ o $\bm{X}_1\perp\bm{X}_2$.

\textit{Aplicaci\'on (Prob.~10.3):} al omitir TFT, el coeficiente de TAM
capturaba su efecto indirecto sobre MI a trav\'es de la fecundidad,
sobreestimando su magnitud en valor absoluto.

%----------------------------------------------------------
\subsection*{A.9\ \ Factor de Inflaci\'on de la Varianza (FIV)}
%----------------------------------------------------------

\[
  \text{FIV}_j=\frac{1}{1-R_j^2},\quad
  \Var(\hat{\beta}_j)=\frac{\hat{\sigma}^2}{\sum x_{ji}^2}\cdot\text{FIV}_j
\]

\begin{center}
\small
\begin{tabular}{cc}
  \toprule
  \rowcolor{softblue}
  FIV & Diagn\'ostico \\
  \midrule
  $1$ & Sin colinealidad \\
  $1$--$5$ & Moderada \\
  $5$--$10$ & Alta \\
  $>10$ & Severa \\
  \bottomrule
\end{tabular}
\end{center}

\textit{Interpretaci\'on:} $\text{FIV}_j=5$ significa que la varianza
de $\hat{\beta}_j$ es 5 veces la que tendr\'ia en ausencia de colinealidad.

%----------------------------------------------------------
\subsection*{A.10\ \ Criterios AIC y BIC}
%----------------------------------------------------------

\[
  \text{AIC}=\ln\!\left(\frac{\text{SRC}}{n}\right)+\frac{2k}{n}
\]
\[
  \text{BIC}=\ln\!\left(\frac{\text{SRC}}{n}\right)+\frac{k\ln n}{n}
\]

BIC penaliza m\'as duramente con muestras grandes ($\ln n>2$).
Se prefiere el modelo con \emph{menor} AIC o BIC.

\textit{Ventaja:} permiten comparar modelos no anidados, algo que $R^2$
no puede hacer al aumentar siempre con m\'as regresores.

\subsection*{A.11\ \ Multicolinealidad: Tipos y Consecuencias}
%----------------------------------------------------------

\textbf{Perfecta:} $\exists\,\bm{c}\neq\bm{0}$: $\bm{Xc}=\bm{0}$.
$\det(\bm{X}'\bm{X})=0$; estimaci\'on imposible.

\textbf{Imperfecta:} correlaci\'on alta pero no exacta.  No viola
supuestos de Gauss-Markov; MCO sigue siendo BLUE.  Consecuencias:
\begin{enumerate}[leftmargin=*,itemsep=0pt,label=\textit{\arabic*.}]
  \item $\Var(\hat{\beta}_j)\uparrow$; estimadores imprecisos
  \item Coeficientes muy sensibles a cambios muestrales
  \item Signos o magnitudes potencialmente inconsistentes con la teor\'ia
  \item Tests $t$ con baja potencia (pero no necesariamente no significativos)
\end{enumerate}

\subsection*{A.12\ \ Determinante e Invertibilidad de $\bm{X}'\bm{X}$}
%----------------------------------------------------------

Para la matriz $2\times2$ de momentos en desviaciones:
\[
  D=\left(\sum x_{2i}^2\right)\!\left(\sum x_{3i}^2\right)-\left(\sum x_{2i}x_{3i}\right)^2
\]

Por Cauchy-Schwarz: $D\geq0$, con igualdad $\Leftrightarrow$ $x_{3i}=\lambda x_{2i}$.

N\'umero de condici\'on: $\kappa=\mu_{\max}/\mu_{\min}$.  Si $\mu_{\min}=0$,
$\kappa=+\infty$ y la matriz es singular.

\textit{Verificaci\'on (Prob.~10.1):}
$D=(110)(440)-(220)^2=48{,}400-48{,}400=0$, $\kappa=550/0=+\infty$.

\subsection*{A.14\ \ Funciones Estimables: Teor\'ia General}
%----------------------------------------------------------

\begin{definicion}
$\bm{\lambda}'\bm{\beta}$ es \textbf{estimable} si existe $\bm{a}$ tal que
$\bm{a}'\bm{y}$ es lineal insesgado de $\bm{\lambda}'\bm{\beta}$.
Equivalentemente: $\bm{\lambda}\in\text{row}(\bm{X})=\text{col}(\bm{X}')$.
\end{definicion}

Cuando $\rg(\bm{X})=r<k$, el espacio estimable tiene dimensi\'on $r$.
Solo $r$ combinaciones lineales independientes pueden estimarse.

\textit{Aplicaci\'on (Prob.~10.2):} con $\rg(\bm{X})=2<3$, solo dos
funciones son estimables: $\alpha_1=\beta_1-\beta_3=-12$ y
$\alpha_2=\beta_2+2\beta_3=2$.  La recta de soluciones
$(\beta_1,\beta_2,\beta_3)=(-12+t,2-2t,t)$ produce siempre el mismo
ajuste $\hat{Y}_i=-12+2X_{2i}$ con $R^2=1$.
